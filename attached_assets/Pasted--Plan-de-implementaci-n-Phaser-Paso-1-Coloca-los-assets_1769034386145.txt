Ejemplo (Phaser 3) listo para pegar
export default class BootScene extends Phaser.Scene {
  constructor() {
    super("BootScene");
  }

  preload() {
    // 1) Carga solo lo mÃ­nimo para el loading UI
    this.load.image("loading_bg", "assets/ui/loading_bg.png");
    this.load.image("bar_frame", "assets/ui/progress_bar_frame.png");
    this.load.image("bar_fill", "assets/ui/progress_bar_fill.png");
    this.load.image("seal", "assets/ui/seal_512.png"); // opcional

    // Cuando termine de cargar lo mÃ­nimo, arrancamos el loader real
    this.load.once("complete", () => this.startRealPreload());
  }

  create() {
    // Nada aquÃ­: esperamos al complete del preload mÃ­nimo
  }

  startRealPreload() {
    const { width, height } = this.scale;

    // --- Fondo ---
    const bg = this.add.image(width / 2, height / 2, "loading_bg");
    bg.setOrigin(0.5);

    // Escala tipo "contain" para que siempre se vea completo
    const scale = Math.min(width / bg.width, height / bg.height);
    bg.setScale(scale);

    // --- Sello opcional ---
    // const seal = this.add.image(width * 0.75, height * 0.55, "seal").setScale(scale * 0.35);

    // --- Barra ---
    const barY = height * 0.78;
    const barFrame = this.add.image(width / 2, barY, "bar_frame");
    barFrame.setOrigin(0.5);

    // Ajusta la barra al mismo scale del fondo
    barFrame.setScale(scale);

    const barFill = this.add.image(barFrame.x, barFrame.y, "bar_fill");
    barFill.setOrigin(0.5);
    barFill.setScale(scale);

    // Crop inicial: 0%
    const fullW = barFill.width;
    const fullH = barFill.height;
    barFill.setCrop(0, 0, 0, fullH);

    // Texto â€œLOADINGâ€¦â€ (si tu BG ya lo tiene, omite esto)
    const loadingText = this.add.text(width / 2, barY - 60 * scale, "LOADING...", {
      fontFamily: "monospace",
      fontSize: `${Math.round(32 * scale)}px`,
      color: "#ffd76a",
    }).setOrigin(0.5);

    // 2) Ahora carga el resto del juego (assets reales)
    // EJEMPLO:
    this.load.image("someSprite", "assets/someSprite.png");
    // ... tu lista completa

    // 3) Progreso real
    this.load.on("progress", (p) => {
      const cropW = Math.floor(fullW * p);
      barFill.setCrop(0, 0, cropW, fullH);
      // opcional: loadingText.setText(`LOADING... ${Math.round(p*100)}%`);
    });

    // 4) Termina y cambia de scene
    this.load.once("complete", () => {
      this.load.off("progress");
      this.scene.start("MenuScene"); // o la que sigue
    });

    this.load.start();
  }
}
