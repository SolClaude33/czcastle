Perfecto: con tu screenshot ya se ve exactamente qué está mal y por qué:

La barra aparece “fuera del frame” porque estás usando setCrop() con un sprite centrado (origin 0.5). En Phaser, cuando recortas un sprite centrado, el recorte se “come” desde ambos lados visualmente y parece que el fill se desalineara.
✅ Solución: el fill debe estar anclado al lado izquierdo del bar (origin X = 0) y posicionado en el borde izquierdo del frame.

El “flash” del loader viejo ocurre porque todavía estás dibujando el loader anterior (rectángulo + glows + textos) mientras los PNG del nuevo loading aún no se muestran (o porque están cargándose y tarda 1 frame en aparecer).
✅ Solución: hacer loading en 2 fases:

Fase A: cargar SOLO assets del loading UI (3 imágenes)

Fase B: mostrar UI y recién ahí cargar el resto del juego con progreso real
Así nunca aparece el loader viejo y la barra siempre está “dentro”.

Abajo te dejo el BootScene final correcto para pegar en game.html. Mantiene tu lista de assets intacta, pero la mueve a una función.

class BootScene extends Phaser.Scene {

  preload() {
    const width = this.cameras.main.width;
    const height = this.cameras.main.height;

    // Fondo fijo inmediato para evitar “flash”
    this.cameras.main.setBackgroundColor('#0a1022');
    this._solid = this.add.rectangle(width/2, height/2, width, height, 0x0a1022);

    // Cargar SOLO lo mínimo del loading UI
    this.load.image('obl_loading_bg', '/assets/ui/loading_bg.png');
    this.load.image('obl_bar_frame', '/assets/ui/progress_bar_frame.png');
    this.load.image('obl_bar_fill', '/assets/ui/progress_bar_fill.png');
  }

  // FASE B: mostrar loading y cargar el juego con progreso real
  create() {
    const width = this.cameras.main.width;
    const height = this.cameras.main.height;

    // --- Dibuja BG (contain 2048x1152 -> 960x540)
    const bg = this.add.image(width/2, height/2, 'obl_loading_bg').setOrigin(0.5);
    const s = Math.min(width / bg.width, height / bg.height);
    bg.setScale(s);

    // (Si tenías el sólido, lo puedes destruir)
    if (this._solid) this._solid.destroy();

    // --- Barra (frame + fill)
    const barY = height * 0.82;

    const barFrame = this.add.image(width/2, barY, 'obl_bar_frame').setOrigin(0.5);
    barFrame.setScale(s);

    // CLAVE: fill anclado a la izquierda del frame
    const leftX = barFrame.x - (barFrame.displayWidth / 2);
    const barFill = this.add.image(leftX, barY, 'obl_bar_fill').setOrigin(0, 0.5);
    barFill.setScale(s);

    // Crop inicial (0)
    const fullW = barFill.width;
    const fullH = barFill.height;
    barFill.setCrop(0, 0, 0, fullH);

    // Opcional: texto LOADING si quieres (si el BG ya lo trae, comenta)
    // const loadingText = this.add.text(width/2, barY - 40*s, 'LOADING...', {
    //   fontFamily: 'monospace',
    //   fontSize: `${Math.round(18*s)}px`,
    //   color: '#ffd76a'
    // }).setOrigin(0.5);

    // Guardar refs para limpiar luego
    this._ui = { bg, barFrame, barFill };

    // ======= Ahora sí, cargar los assets del juego (tu lista completa) =======
    this.queueGameAssets(); // <- aquí metemos tu lista
    this.load.on('progress', (p) => {
      // crop desde izquierda, sin desplazamiento raro
      const cropW = Math.floor(fullW * p);
      barFill.setCrop(0, 0, cropW, fullH);
    });

    this.load.once('complete', () => {
      // limpiar listeners y UI
      this.load.off('progress');
      // if (loadingText) loadingText.destroy();
      this._ui.bg.destroy();
      this._ui.barFrame.destroy();
      this._ui.barFill.destroy();

      // Animaciones + flujo real
      this.createAnimations();
      this.scene.start('MenuScene');
    });

    // Arranca el loader real
    this.load.start();
  }
