class BootScene extends Phaser.Scene {
  constructor() {
    super('BootScene');

    // refs internas
    this._loadingUI = null;
    this._progressHandler = null;

    this._lastProgress = 0;

  preload() {
    const width = this.cameras.main.width;
    const height = this.cameras.main.height;

    // Fondo oscuro mientras aparecen los PNG del loading
    this.cameras.main.setBackgroundColor('#0a1022');

    // ========= (A) UI del LOADING (mínimo) =========
    // Se cargan ANTES que todo para poder mostrarlas durante el preload
    this.load.image('obl_loading_bg', 'assets/ui/loading_bg.png');
    this.load.image('obl_bar_frame', 'assets/ui/progress_bar_frame.png');
    this.load.image('obl_bar_fill', 'assets/ui/progress_bar_fill.png');
    // opcional:
    // this.load.image('obl_seal', 'assets/ui/seal_512.png');

    // Creamos la UI apenas estén los 3 assets clave del loading
    const uiFlags = { bg: false, frame: false, fill: false };
    const tryBuildUI = () => {
      if (this._loadingUI) return; // ya creada
      if (!uiFlags.bg || !uiFlags.frame || !uiFlags.fill) return;

      // --- BACKGROUND (contain en 960x540) ---
      const bg = this.add.image(width / 2, height / 2, 'obl_loading_bg').setOrigin(0.5);
      const s = Math.min(width / bg.width, height / bg.height);
      bg.setScale(s);

      // --- PROGRESS BAR (centrada abajo) ---
      const barY = height * 0.82;

      const barFrame = this.add.image(width / 2, barY, 'obl_bar_frame').setOrigin(0.5);
      barFrame.setScale(s);

      const barFill = this.add.image(width / 2, barY, 'obl_bar_fill').setOrigin(0.5);
      barFill.setScale(s);

      // Crop en espacio de textura (antes de escala)
      const fullW = barFill.width;
      const fullH = barFill.height;
      barFill.setCrop(0, 0, 0, fullH);

      // Texto opcional (si el bg ya trae LOADING, comenta esto)
      const loadingText = this.add.text(width / 2, barY - 40 * s, 'LOADING...', {
        fontFamily: 'monospace',
        fontSize: `${Math.round(18 * s)}px`,
        color: '#ffd76a'
      }).setOrigin(0.5);

      // Guarda refs
      this._loadingUI = { bg, barFrame, barFill, loadingText, fullW, fullH };

      // Aplica el progreso actual por si ya avanzó antes de crear la UI
      const cropW = Math.floor(fullW * this._lastProgress);
      barFill.setCrop(0, 0, cropW, fullH);
    };

    // File-complete listeners (solo para construir UI del loading)
    this.load.on('filecomplete-image-obl_loading_bg', () => { uiFlags.bg = true; tryBuildUI(); });
    this.load.on('filecomplete-image-obl_bar_frame',  () => { uiFlags.frame = true; tryBuildUI(); });
    this.load.on('filecomplete-image-obl_bar_fill',   () => { uiFlags.fill = true; tryBuildUI(); });

    // ========= (B) Progreso REAL (para TODA la carga) =========
    this._progressHandler = (value) => {
      this._lastProgress = value;
      if (!this._loadingUI) return;

      const { barFill, fullW, fullH } = this._loadingUI;
      const cropW = Math.floor(fullW * value);
      barFill.setCrop(0, 0, cropW, fullH);
    };

    this._completeHandler = () => {
      // Limpia listeners para evitar leaks
      this.load.off('progress', this._progressHandler);
      this.load.off('complete', this._completeHandler);

      // (Opcional) también puedes remover los filecomplete si quieres
      // this.load.off('filecomplete-image-obl_loading_bg');
      // ...

      // Destruye UI del loading antes de pasar a MenuScene
      if (this._loadingUI) {
        this._loadingUI.loadingText?.destroy();
        this._loadingUI.barFill?.destroy();
        this._loadingUI.barFrame?.destroy();
        this._loadingUI.bg?.destroy();
        this._loadingUI = null;
      }
    };

    this.load.on('progress', this._progressHandler);
    this.load.once('complete', this._completeHandler);

    // ========= (C) TU LISTA DE ASSETS (la dejas igual) =========
    this.load.image('tilemap-color1', ASSET_BASE + 'Terrain/Tileset/Tilemap_color1.png');
    this.load.image('tilemap-color2', ASSET_BASE + 'Terrain/Tileset/Tilemap_color2.png');

    this.load.spritesheet('warrior-blue-run', ASSET_BASE + 'Units/Blue Units/Warrior/Warrior_Run.png', { frameWidth: 192, frameHeight: 192 });
