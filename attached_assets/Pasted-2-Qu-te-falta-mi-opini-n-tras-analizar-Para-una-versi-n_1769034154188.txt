C) Preloader real (src/preload.ts)
export function preloadImages(
  urls: readonly string[],
  onProgress: (p: number) => void
) {
  let loaded = 0;

  return Promise.all(
    urls.map(
      (url) =>
        new Promise<void>((resolve) => {
          const img = new Image();
          const done = () => {
            loaded += 1;
            onProgress(loaded / urls.length);
            resolve();
          };
          img.onload = done;
          img.onerror = done;
          img.src = url;
        })
    )
  );
}

E) App con loading real (src/App.tsx)
import { useEffect, useState } from "react";
import { ASSETS } from "./assets";
import { preloadImages } from "./preload";
import { LoadingScreen } from "./LoadingScreen";
import Home from "./Home";

export default function App() {
  const [progress, setProgress] = useState(0);
  const [ready, setReady] = useState(false);

  useEffect(() => {
    let alive = true;
    const start = performance.now();

    (async () => {
      await preloadImages(ASSETS, (p) => alive && setProgress(p));

      // evita “flash” si carga instantáneo:
      const minTime = 700;
      const elapsed = performance.now() - start;
      if (elapsed < minTime) await new Promise(r => setTimeout(r, minTime - elapsed));

      if (alive) setReady(true);
    })();

    return () => { alive = false; };
  }, []);

  if (!ready) return <LoadingScreen progress={progress} />;
  return <Home />;
}
